      }

      onBridgeReady(res.data);

      WeixinJSBridge.invoke(
        'getBrandWCPayRequest',

          "signType": payParams.signType,

          if (res.err_msg === "get_brand_wcpay_request:ok") {
            // 支付成功
            window.location.href = '/order-success.html?orderId=' + orderId;
          } else {
            alert('支付失败，请重试');
          }
        }
      );
    }
  } catch (error) {
    console.error('支付失败:', error);

```javascript
// server/routes/pay.js - generateAlipaySign函数
function generateAlipaySign(params, privateKey) {
  // 参数排序并拼接
  const sorted = Object.keys(params).sort().map(k => `${k}=${params[k]}`).join('&');
  
  // 使用RSA2私钥签名
  const crypto = require('crypto');
  const sign = crypto.createSign('RSA-SHA256');
  sign.update(sorted);
  
  // 确保私钥格式正确（需包含头尾）
  const formattedKey = privateKey.includes('-----BEGIN')
    ? privateKey
    : `-----BEGIN RSA PRIVATE KEY-----\n${privateKey}\n-----END RSA PRIVATE KEY-----`;
  
  return sign.sign(formattedKey, 'base64');
}
```

  // 获取支持的快递公司
  async getCarriers() {
    return await apiRequest('/shipping/carriers');
  }
};

## 十、常见问题FAQ

### Q1: 微信支付提示"商家参数格式有误"
**原因**：支付授权目录配置不正确  
**解决**：在商户平台 → 开发配置 → 支付授权目录，设置为实际访问域名

### Q2: 支付宝支付页面空白
**原因**：签名格式不正确或公私钥不匹配  
**解决**：检查密钥格式，确保使用RSA2签名

### Q3: 快递100轨迹查询失败
**原因**：参数不完整或快递公司代码错误  
**解决**：确认快递公司代码在支持列表中，phone参数可选但建议提供

### Q4: 支付回调验签失败
**原因**：密钥不匹配或参数顺序错误  
**解决**：仔细核对商户平台密钥，确保参数排序正确

### Q5: 本地测试支付功能
**方案A**：使用微信/支付宝沙箱环境  
**方案B**：使用内网穿透工具（如natapp、ngrok）暴露本地端口
